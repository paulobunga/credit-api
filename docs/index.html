    Authenticating requests

    This API is authenticated by sending a parameter sign in the body or query of the request.

    All authenticated endpoints are marked with a requires authentication badge in the documentation below.

    You can create the sign value by the following Pseudo code.

        
    /**
     * @param string $key The API Key of merchant
     * @param array  $data The json data of Http request
     * 
     * /
    protected function createSign(array $data, String $key)
    {
        ksort($data);
        $str = "";
        foreach ($data as $key => $val) {
            if ($key == "sign") {
                continue;
            }
            if (is_null($val) || $val === "") {
                continue;
            }
            if (is_array($val) || is_object($val)) {
                $val = json_encode($val);
            }
            $str .= "{$key}={$val}&";
        }
        $str .= "api_key=" .  $key;

        return md5($str);
    }

    Get deposit list

    requires authentication

    This endpoint lets you get deposit list.

    Example request:
    curl --request GET \
    --get "api.credit-api.test/deposits?uuid=224d4a1f-6fc5-4039-bd81-fcbc7f88c659&sign=e38c3a02a3d9757c912d0dc6240a5c88" \
    --header "Api-Version: v1" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json"
    $client = new \GuzzleHttp\Client();
$response = $client->get(
    'api.credit-api.test/deposits',
    [
        'headers' => [
            'Api-Version' => 'v1',
            'Accept' => 'application/json',
        ],
        'query' => [
            'uuid'=> '224d4a1f-6fc5-4039-bd81-fcbc7f88c659',
            'sign'=> 'e38c3a02a3d9757c912d0dc6240a5c88',
        ],
    ]
);
$body = $response->getBody();
print_r(json_decode((string) $body));

    Create a deposit

    requires authentication

    This endpoint lets you create a deposit.

    Example request:
    curl --request POST \
    "api.credit-api.test/deposits" \
    --header "Api-Version: v1" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"merchant_order_id\": \"9798223690986\",
    \"currency\": \"VND\",
    \"channel\": \"MOMOPAY\",
    \"method\": \"TEXT\",
    \"uuid\": \"224d4a1f-6fc5-4039-bd81-fcbc7f88c659\",
    \"sign\": \"d72b7a2c1fda1b4914354e6c5dbb0038\",
    \"amount\": \"1000\",
    \"callback_url\": \"http:\\/\\/callback.url\\/0001\"
}"

    $client = new \GuzzleHttp\Client();
$response = $client->post(
    'api.credit-api.test/deposits',
    [
        'headers' => [
            'Api-Version' => 'v1',
            'Accept' => 'application/json',
        ],
        'json' => [
            'merchant_order_id' => '9798223690986',
            'currency' => 'VND',
            'channel' => 'MOMOPAY',
            'method' => 'TEXT',
            'uuid' => '224d4a1f-6fc5-4039-bd81-fcbc7f88c659',
            'sign' => 'd72b7a2c1fda1b4914354e6c5dbb0038',
            'amount' => '1000',
            'callback_url' => 'http://callback.url/0001',
        ],
    ]
);
$body = $response->getBody();
print_r(json_decode((string) $body));
